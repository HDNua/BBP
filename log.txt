* !!!!! IMPORTANT & HARD CHALLENGE !!!!!
  지금 버전에서는 사각형의 모양이 엄청나게 특수하지는 않다는 점을 들어서,
  평행 사변형 정도나 한 변이 직각인 사다리꼴만 있다고 간주한다.
  그러면 다음과 같은 정렬 방법으로 정렬된 배열을 얻을 수 있다.

  List<Vector2> sorted = new List<Vector2>(points);
  sorted.Sort(delegate (Vector2 a, Vector2 b)
  {
    if (a.x != b.x)
    {
      return (a.x < b.x) ? -1 : 1;
    }
    else if (a.y != b.y)
    {
      return (a.y > b.y) ? -1 : 1;
    }
    return 0;
  });

  ret[0] = sorted[0];
  ret[1] = sorted[2];
  ret[2] = sorted[1];
  ret[3] = sorted[3];
  return ret;
